**Goal:** Enable step response tests over RS485 serial so the dashboard's PID tuning tab works without ROS2 nodes — completing the read→test→measure→tune→compare pipeline.

**Architecture:** Add `_rs485_run_step_test()` to `PIDTuningBridge` that sends a position command via the existing RS485 driver, polls motor state in a background thread at ~10 Hz, collects time-series data, computes metrics using the existing `step_metrics.py` (pure NumPy), and returns results in the exact format the frontend expects. The RS485 step test replicates the C++ action server logic:
1. Read initial position via `read_multi_turn_angle()`
2. Compute target = initial + step_size_degrees
3. Send position command via `send_position_command(target_centideg, max_speed=0)`
4. Poll at ~10 Hz for `duration` seconds, collecting: timestamp, position (degrees),
   speed (dps from status2), current (amps from status2)
5. After collection, stop motor (or hold at target)
6. Compute metrics via `step_metrics.compute_step_metrics()`
7. Return `{success, timestamps, positions, velocities, currents, setpoint, metrics}`

**Step 1.3: Implement `_rs485_run_step_test()` in PIDTuningBridge**

Add to `PIDTuningBridge` class in `pid_tuning_api.py`:

```python
# --- Constants for RS485 step test ---
RS485_STEP_TEST_SAMPLE_RATE_HZ = 10.0
RS485_STEP_TEST_TEMP_LIMIT_C = 80.0
RS485_STEP_TEST_MAX_DEVIATION_FACTOR = 2.0  # abort if deviation > 2x step size

def _rs485_run_step_test(
    self,
    motor_id: int,
    step_size: float,
    duration: float,
) -> dict:
    """Execute step response test over RS485 serial.

    Replicates the C++ action server logic:
    1. Read initial position
    2. Command target position (initial + step_size)
    3. Poll motor state at ~10 Hz for duration seconds
    4. Compute step response metrics
    5. Return time-series + metrics in same format as ROS2 path
    """
    driver = self._rs485_driver
    if driver is None:
        return {"success": False, "error": "RS485 driver not available"}

    # 1. Read initial position
    initial_angle = driver.read_multi_turn_angle()
    if initial_angle is None:
        return {"success": False, "error": "Failed to read initial angle"}

    # 2. Compute and command target position
    target_angle = initial_angle + step_size
    target_centideg = int(round(target_angle * 100))

    cmd_result = driver.send_position_command(target_centideg, max_speed_dps=0)
    if cmd_result is None:
        return {"success": False, "error": "Failed to send position command"}

    # 3. Poll at ~10 Hz, collect time-series
    timestamps = []
    positions = []
    velocities = []
    currents = []

    sample_period = 1.0 / RS485_STEP_TEST_SAMPLE_RATE_HZ
    max_deviation_deg = abs(step_size) * RS485_STEP_TEST_MAX_DEVIATION_FACTOR
    start_time = time.monotonic()

    try:
        while True:
            elapsed = time.monotonic() - start_time
            if elapsed >= duration:
                break

            # Read position
            angle = driver.read_multi_turn_angle()
            if angle is None:
                continue  # skip failed reads

            # Read status2 for speed/current/temp
            status = driver.read_status_2()
            if status is None:
                continue

            sample_time = time.monotonic() - start_time
            timestamps.append(sample_time)
            positions.append(angle)
            velocities.append(status["speed_dps"])
            currents.append(status["torque_current_a"])

            # Safety: temperature check
            if status["temperature_c"] > RS485_STEP_TEST_TEMP_LIMIT_C:
                driver.motor_stop()
                return {
                    "success": False,
                    "error": f"Motor temperature {status['temperature_c']}C "
                             f"exceeds limit {RS485_STEP_TEST_TEMP_LIMIT_C}C",
                    "timestamps": timestamps,
                    "positions": positions,
                    "velocities": velocities,
                    "currents": currents,
                    "setpoint": target_angle,
                }

            # Safety: position deviation check
            if max_deviation_deg > 0.01:
                deviation = abs(angle - target_angle)
                if deviation > max_deviation_deg:
                    driver.motor_stop()
                    return {
                        "success": False,
                        "error": f"Position deviation {deviation:.1f} deg "
                                 f"exceeds limit {max_deviation_deg:.1f} deg",
                        "timestamps": timestamps,
                        "positions": positions,
                        "velocities": velocities,
                        "currents": currents,
                        "setpoint": target_angle,
                    }

            # Sleep until next sample
            next_sample = start_time + len(timestamps) * sample_period
            sleep_time = next_sample - time.monotonic()
            if sleep_time > 0:
                time.sleep(sleep_time)

    except Exception as exc:
        driver.motor_stop()
        return {"success": False, "error": f"Step test error: {exc}"}

    if len(timestamps) < 2:
        return {"success": False, "error": "Insufficient data collected"}

    # 4. Compute metrics
    result = {
        "success": True,
        "timestamps": timestamps,
        "positions": positions,
        "velocities": velocities,
        "currents": currents,
        "setpoint": target_angle,
    }

    try:
        import numpy as np
        from pid_tuning.step_metrics import compute_step_metrics

        metrics_result = compute_step_metrics(
            timestamps=np.array(timestamps),
            positions=np.array(positions),
            setpoint=target_angle,
            step_size=abs(step_size),
        )
        result["metrics"] = {
            "rise_time": metrics_result.rise_time,
            "settling_time": metrics_result.settling_time,
            "overshoot_percent": metrics_result.overshoot_percent,
            "steady_state_error": metrics_result.steady_state_error,
            "iae": metrics_result.iae,
            "ise": metrics_result.ise,
            "itse": metrics_result.itse,
            "data_points": metrics_result.data_points,
            "confidence": metrics_result.confidence,
        }
    except ImportError:
        logger.warning("step_metrics not available, skipping metric computation")
    except Exception:
        logger.exception("Failed to compute step metrics")

    return result
```
